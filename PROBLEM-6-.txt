(STRCPY)

#include <stdio.h>
#include <stdlib.h>
#define CRT_SECURE_NO_WARNINGS

int main() {
	char sentence[50] = "Hello World!";  //write how many characters will be
	char copy[50] = " "; //not put data in here

	strcpy(copy, sentence);  //write where the first will be copied, the second where to copy
	printf("%s", copy);   //to copy and displays the data in sentence

	return 0;
}


(STRCAT)

#include <stdio.h>
#include <string.h>

int main() {
	char country[100] = "Turkey", capital[] = "Ankara";  //characters are defined to char

	strcat(country, capital);   //written sequentially to combine the two

	puts(country);  //to print on the screen
	puts(capital);

	return 0;
		
}
